SingleElementSetAlgebra( FiniteAlgebra ) -- algebra with single set of elements
    about -- prints info about the algebra
    deepcopy
    dump -- saves algebra to file in JSON format
    dumps -- outputs algebra as JSON string
    elements -- list of names of the algebra's elements (strings)
    has_identity -- returns True if the algebra has an identity
    has_inverses -- returns True if the algebra has inverses
    identity -- returns the algebra's identity element, if it exists
    inv -- returns and element's inverse element
    is_abelian -- returns True if the algebra is abelian
    is_associative -- returns True if the algebra is associative
    is_commutative -- same as is_abelian
    order -- returns the order of the algebra
    table -- returns the algebra's Cayley table
    to_dict -- outputs the algebra as a Python dictionary
